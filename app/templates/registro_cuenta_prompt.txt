        time.sleep(0.1)

# Campo Email

    driver.execute_script("""
        arguments[0].style.border = '3px solid green';

try:
    password_field = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.XPATH, "//input[@type='password' and (@name='password' or @id='password' or contains(@placeholder, 'contrase√±a'))]"))

# Buscar mensaje de √©xito
success_found = False

if success_found or "dashboard" in current_url.lower() or "home" in current_url.lower():
