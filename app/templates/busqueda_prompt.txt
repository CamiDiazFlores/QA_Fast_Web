url = "{url}"

print("=" * 80)

# Buscar campo de b√∫squeda
print("\nüìç PASO 2: Localizando campo de b√∫squeda")
try:
    search_field = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.XPATH, "//input[@type='search'] | //input[@name='search'] | //input[@id='search'] | //input[@placeholder*='buscar'] | //input[@placeholder*='search']"))
    )
    
    driver.execute_script("""
        arguments[0].style.border = '5px solid blue';
        arguments[0].style.backgroundColor = '#e3f2fd';
        arguments[0].scrollIntoView({{block: 'center'}});
    """, search_field)
    time.sleep(2)
    
    print("‚úÖ Campo de b√∫squeda encontrado")
    
    # Escribir t√©rmino
    print(f"\nüìç PASO 3: Escribiendo t√©rmino '{{search_term}}'")
    search_field.clear()
    for char in search_term:
        search_field.send_keys(char)
        time.sleep(0.15)
    
    print(f"‚úÖ T√©rmino ingresado: {{search_term}}")
    time.sleep(2)
    
    # Buscar bot√≥n de b√∫squeda
    try:
        search_button = driver.find_element(By.XPATH, "//button[@type='submit'] | //button[contains(., 'Buscar')] | //button[contains(., 'Search')]")
        driver.execute_script("arguments[0].style.border='5px solid green';", search_button)
        time.sleep(1)
        search_button.click()
        print("‚úÖ Bot√≥n de b√∫squeda clickeado")
    except:
        search_field.send_keys(Keys.RETURN)
        print("‚úÖ Enter enviado")
    
    time.sleep(5)
    
except Exception as e:
    print(f"‚ùå Error: {{e}}")
    save_screenshot()

# Validar resultados
print("\nüìç PASO 4: Validando resultados de b√∫squeda")
time.sleep(3)

try:
    results = driver.find_elements(By.XPATH, "//*[contains(@class, 'result')] | //*[contains(@class, 'search-result')] | //article | //div[contains(@class, 'item')]")
    
    results_count = len(results)
    print(f"‚úÖ Resultados encontrados: {{results_count}}")
    
    if results_count > 0:
        # Resaltar primer resultado
        driver.execute_script("""
            arguments[0].style.border = '5px solid green';
            arguments[0].style.backgroundColor = '#c8e6c9';
            arguments[0].scrollIntoView({{block: 'center'}});
        """, results[0])
        
        print("=" * 80)
        print("‚úÖ TEST EXITOSO: B√öSQUEDA FUNCIONANDO")
        print(f"‚úÖ {{results_count}} resultados encontrados")
        print("=" * 80)
        
        save_screenshot()
        time.sleep(3)
    else:
        print("‚ùå No se encontraron resultados")
        save_screenshot()
        
except Exception as e:
    print(f"‚ö†Ô∏è Error al validar resultados: {{e}}")
    save_screenshot()
