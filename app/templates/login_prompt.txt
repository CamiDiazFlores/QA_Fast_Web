# Buscar campo usuario
print("\nüìç PASO 2: Buscando campo de usuario")
try:
    username_field = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.XPATH, "//input[@type='text'] | //input[@name='username'] | //input[@id='username']"))
    )
    
    # Resaltar campo
    driver.execute_script("""
        arguments[0].style.border = '3px solid orange';
        arguments[0].style.backgroundColor = '#fff3cd';
        arguments[0].scrollIntoView({{block: 'center'}});
    """, username_field)
    time.sleep(1)
    
    print("‚úÖ Campo de usuario encontrado")
    
    # Escribir usuario letra por letra
    username_field.clear()
    for char in username:
        username_field.send_keys(char)
        time.sleep(0.15)
    
    print(f"‚úÖ Usuario ingresado: {{username}}")
    time.sleep(2)
    
except Exception as e:
    print(f"‚ùå Error: {{e}}")
    save_screenshot()

# Buscar campo contrase√±a
print("\nüìç PASO 3: Buscando campo de contrase√±a")
try:
    password_field = WebDriverWait(driver, 10).until(
        EC.presence_of_element_located((By.XPATH, "//input[@type='password']"))
    )
    
    driver.execute_script("""
        arguments[0].style.border = '3px solid red';
        arguments[0].style.backgroundColor = '#f8d7da';
        arguments[0].scrollIntoView({{block: 'center'}});
    """, password_field)
    time.sleep(1)
    
    print("‚úÖ Campo de contrase√±a encontrado")
    
    # Escribir contrase√±a (con asteriscos visuales)
    for i, char in enumerate(password):
        password_field.send_keys(char)
        time.sleep(0.15)
        print(f"üîí Escribiendo... {{'*' * (i+1)}}")
    
    print("‚úÖ Contrase√±a ingresada")
    time.sleep(2)
    
except Exception as e:
    print(f"‚ùå Error: {{e}}")
    save_screenshot()

# Buscar bot√≥n login
print("\nüìç PASO 4: Buscando bot√≥n de login")
try:
    login_button = WebDriverWait(driver, 10).until(
        EC.element_to_be_clickable((By.XPATH, "//button[@type='submit'] | //button[contains(text(), 'Login')] | //input[@type='submit']"))
    )
    
    driver.execute_script("""
        arguments[0].style.border = '5px solid green';
        arguments[0].scrollIntoView({{block: 'center'}});
    """, login_button)
    time.sleep(1)
    
    print("‚úÖ Bot√≥n de login encontrado")
    login_button.click()
    print("‚úÖ Clic en bot√≥n de login")
    time.sleep(5)
    
except Exception as e:
    print(f"‚ùå Error: {{e}}")
    save_screenshot()

if "dashboard" in current_url.lower() or "home" in current_url.lower():
    print("‚úÖ LOGIN EXITOSO")

        var div = document.createElement('div');
        div.innerHTML = '‚úÖ LOGIN EXITOSO';
